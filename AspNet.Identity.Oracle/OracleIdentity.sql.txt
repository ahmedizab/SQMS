CREATE TABLE "ASPNETUSERCLAIMS"
(
	"ID"     NUMBER(10, 0) NOT NULL,
	"USERID" VARCHAR2(128 BYTE) NOT NULL,
	"CLAIMTYPE" CLOB,
	"CLAIMVALUE" CLOB
);
CREATE UNIQUE INDEX "PK_ASPNETUSERCLAIMS_ID" ON "ASPNETUSERCLAIMS" ("ID");
ALTER TABLE "ASPNETUSERCLAIMS" ADD CONSTRAINT "PK_ASPNETUSERCLAIMS_ID" PRIMARY KEY ("ID");

CREATE INDEX "FK_ASPNETUSERCLAIMS_USERID" ON "ASPNETUSERCLAIMS" ("USERID");
ALTER TABLE "ASPNETUSERCLAIMS" ADD CONSTRAINT "FK_ASPNETUSERCLAIMS_USERID" FOREIGN KEY ("USERID")
  REFERENCES "ASPNETUSERS" ("ID") ON DELETE CASCADE ENABLE;

CREATE SEQUENCE "SEQ_ASPNETUSERCLAIMS_ID"
START WITH 1
INCREMENT BY 1
NOMAXVALUE; 

CREATE OR REPLACE TRIGGER "TR_ASPNETUSERCLAIMS_INS" 
	BEFORE INSERT
	ON "ASPNETUSERCLAIMS"
	REFERENCING NEW AS new OLD AS old
	FOR EACH ROW
BEGIN
	IF INSERTING
	THEN
		IF :new.id IS NULL
		THEN
			SELECT SEQ_ASPNETUSERCLAIMS_ID.NEXTVAL INTO :new.id FROM DUAL;
		END IF;
	END IF;
END;
/
ALTER TRIGGER "TR_ASPNETUSERCLAIMS_INS" ENABLE;
